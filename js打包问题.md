dataurl = getSendUrl();

e = 0;
while (i--) {
    jQuery.ajax({
        type: "post",
        url: "/Servlet/recordStudy.svl" + "?" + dataurl,

        cache: false,
        success: function (a) {
            onSendRecordSuccess(a);
